pipeline
{
	
	agent any
	stages
	{
		stages ('Code Checkout')
		{
			git 'https://github.com/ershekhar007/igp-test.git'
		}
	}

	stage ('Code Compile')
	{
		steps
		{
			sh 'mvn compile'
		}
	}

	stage ('Unit Test')
	{
		steps
		{
			sh 'mvn test'
		}
	}

	stage ('Code Packaging')
	{
		steps
		{
			sh 'mvn package'
		}
	}
}
}
